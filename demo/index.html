Code Example

import { QSticker, createSticker } from 'qsticker';

// Create a sticker from template
const sticker = createSticker('star', {
  width: 150,
  height: 150,
  color: '#FFD700',
  animation: {
    type: 'pulse',
    duration: 2000,
    iterationCount: 'infinite'
  }
});

// Render to container
sticker.render('#myContainer');

// Update properties
sticker.setText('NEW!');
sticker.setColor('#FF69B4');
sticker.setAnimation({ type: 'bounce' });

// Export options
const dataURL = sticker.toDataURL();
const blob = sticker.toBlob();
const image = await sticker.toImage();
